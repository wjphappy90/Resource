Los atributos se especifican con el símbolo prefijo '@'.
 
//@id
Selecciona todos los atributos 'id'

     <AAA>
          <BBB id = "b1"/>
          <BBB id = "b2"/>
          <BBB name = "bbb"/>
          <BBB/>
     </AAA>

//BBB[@id]
Selecciona los elementos BBB que contienen un atributo 'id'

     <AAA>
          <BBB id = "b1"/>
          <BBB id = "b2"/>
          <BBB name = "bbb"/>
          <BBB/>
     </AAA>
