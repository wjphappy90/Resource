Se il path inizia con // allora vengono selezionati tutti gli elementi del documento che soddisfano i criteri successivi.
 
//BBB
Selezione di tutti gli elementi BBB

     <AAA>
          <BBB/>
          <CCC/>
          <BBB/>
          <DDD>
               <BBB/>
          </DDD>
          <CCC>
               <DDD>
                    <BBB/>
                    <BBB/>
               </DDD>
          </CCC>
     </AAA>
