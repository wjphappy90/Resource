import java.util.Scanner;

/**
 * 工龄[10,15]   +5000
 *    [5,10)    +2500
 *    [3,5)     +1000
 *    [1,3)     +500
 *    [0,1)     +200
 * @author JX
 *
 */
public class Test5 {
	public static void main(String[] args) {
		//1.定义键盘录入对象
		Scanner sc = new Scanner(System.in);
		System.out.println("请输入基本工资(整数)：");
		//2.输入员工基本工资
		int salary = sc.nextInt();
		System.out.println("请输入工龄(整年)：");
		//3.输入工龄
		int workAge = sc.nextInt();
		//4.定义临时变量tempSalary用于存储所涨的工资
		int tempSalary = 0;
		//5.用if判断给员工涨工资
		if(workAge<0||workAge>15) {
			System.out.println("您输入的工龄有误！");
			return ;
		} else if(workAge>=10) {
			tempSalary = 5000;
		} else if(workAge>=5) {
			tempSalary = 2500;
		} else if(workAge>=3) {
			tempSalary = 1000;
		} else if(workAge>=1) {
			tempSalary = 500;
		} else {
			tempSalary = 200;
		}
		System.out.println("您目前工作了"+workAge+"年，基本工资为 "+salary+", 应涨工资 "+tempSalary+"元,涨后工资 "+(salary+tempSalary));
	}
}
